# /rottengenizdat Slack Bot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Cloudflare Worker that handles `/rottengenizdat` slash command with a modal UI for picking audio recipes, then dispatches a GitHub Actions workflow.

**Architecture:** Cloudflare Worker receives Slack slash commands and interactions. The modal lets users pick a recipe, sample count, and input mode. On submit, it dispatches a `workflow_dispatch` to `rottengenizdat.yml` which runs the audio pipeline and posts results back to Slack. Modeled directly on the sparagmos worker but stripped of KV (no ratings/voting/favorites).

**Tech Stack:** TypeScript (Cloudflare Workers), Slack Block Kit, GitHub Actions API, Vitest

```typescript
import type { ParsedCommand } from "./types";

// ---------------------------------------------------------------------------
// Slash command parsing
// ---------------------------------------------------------------------------

/**
 * Parse a Slack slash command text field into a command and optional URLs.
 *
 * The first non-URL token is the command (lowercased). All tokens matching
 * http(s):// are collected as URLs with original case preserved.
 */
export function parseSlashCommand(text: string): ParsedCommand {
  const parts = text.trim().split(/\s+/).filter(Boolean);
  if (parts.length === 0) {
    return { command: "", urls: [] };
  }

  const first = parts[0];
  const isUrl = /^https?:\/\//i;

  if (isUrl.test(first)) {
    return {
      command: "",
      urls: parts.filter((s) => isUrl.test(s)),
    };
  }

  return {
    command: first.toLowerCase(),
    urls: parts.slice(1).filter((s) => isUrl.test(s)),
  };
}

// ---------------------------------------------------------------------------
// Slack signature verification
// ---------------------------------------------------------------------------

/**
 * Verify the Slack request signature using HMAC-SHA256.
 *
 * Checks the `x-slack-signature` and `x-slack-request-timestamp` headers
 * against the raw request body. Rejects requests older than 5 minutes to
 * prevent replay attacks.
 */
export async function verifySlackSignature(
  request: Request,
  body: string,
  signingSecret: string,
): Promise<boolean> {
  const timestamp = request.headers.get("x-slack-request-timestamp");
  const slackSignature = request.headers.get("x-slack-signature");

  if (!timestamp || !slackSignature) {
    return false;
  }

  const now = Math.floor(Date.now() / 1000);
  const ts = Number(timestamp);
  if (Number.isNaN(ts) || Math.abs(now - ts) > 300) {
    return false;
  }

  const sigBasestring = `v0:${timestamp}:${body}`;

  const encoder = new TextEncoder();
  const key = await crypto.subtle.importKey(
    "raw",
    encoder.encode(signingSecret),
    { name: "HMAC", hash: "SHA-256" },
    false,
    ["sign"],
  );

  const signatureBytes = await crypto.subtle.sign(
    "HMAC",
    key,
    encoder.encode(sigBasestring),
  );

  const expectedSignature =
    "v0=" +
    [...new Uint8Array(signatureBytes)]
      .map((b) => b.toString(16).padStart(2, "0"))
      .join("");

  if (expectedSignature.length !== slackSignature.length) {
    return false;
  }
  let mismatch = 0;
  for (let i = 0; i < expectedSignature.length; i++) {
    mismatch |= expectedSignature.charCodeAt(i) ^ slackSignature.charCodeAt(i);
  }
  return mismatch === 0;
}

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

/** Build a JSON response that Slack understands. */
export function slackResponse(text: string, ephemeral = true): Response {
  return new Response(
    JSON.stringify({
      response_type: ephemeral ? "ephemeral" : "in_channel",
      text,
    }),
    {
      headers: { "Content-Type": "application/json" },
    },
  );
}
```

### Task 3: Recipe Generation Script

**Files:**
- Create: `scripts/generate_worker_recipes.py`

This adapts the sparagmos script from YAML to TOML. Rottengenizdat recipes use `[recipe]` table with `name` and `mode`, plus `[[steps]]` arrays with `effect`, `model`, `temperature`, etc.

- [ ] **Step 1: Create `scripts/generate_worker_recipes.py`**

```python
#!/usr/bin/env python3
"""Generate worker/src/recipes.ts from recipes/*.toml.

Reads all TOML recipe files and generates a TypeScript module that the
Cloudflare Worker uses for recipe validation, listing, and suggestion.

Usage:
    python scripts/generate_worker_recipes.py
"""

from __future__ import annotations

import argparse
import re
import sys
from pathlib import Path

try:
    import tomllib
except ModuleNotFoundError:
    import tomli as tomllib  # type: ignore[no-redef]


def find_repo_root() -> Path:
    """Find the repo root by looking for the recipes/ directory."""
    script_dir = Path(__file__).resolve().parent
    repo_root = script_dir.parent
    if (repo_root / "recipes").is_dir():
        return repo_root
    if (Path.cwd() / "recipes").is_dir():
        return Path.cwd()
    return repo_root


def escape_ts_string(s: str) -> str:
    """Escape a string for use in a TypeScript double-quoted string literal."""
    return s.replace("\\", "\\\\").replace('"', '\\"').replace("\n", "\\n")


def annotate_step(step: dict) -> str:
    """Return an annotated label for a single recipe step.

    Examples:
        {"effect": "rave", "model": "percussion", "temperature": 1.2}
        -> "rave(percussion t=1.2)"

        {"effect": "dry", "weight": 0.9}
        -> "dry(w=0.9)"
    """
    effect = step.get("effect", "unknown")
    parts: list[str] = []
    if "model" in step:
        parts.append(step["model"])
    if "temperature" in step:
        parts.append(f"t={step['temperature']}")
    if "noise" in step:
        parts.append(f"n={step['noise']}")
    if "weight" in step:
        parts.append(f"w={step['weight']}")
    if step.get("reverse"):
        parts.append("rev")
    if "shuffle_chunks" in step:
        parts.append(f"shuf={step['shuffle_chunks']}")
    if "quantize" in step:
        parts.append(f"q={step['quantize']}")
    if "dims" in step:
        parts.append(f"d={step['dims']}")
    if "mix" in step:
        parts.append(f"mix={step['mix']}")
    if parts:
        return f"{effect}({' '.join(parts)})"
    return effect


def build_effects_chain(steps: list[dict], mode: str) -> str:
    """Build a human-readable effect chain string from recipe steps."""
    sep = " + " if mode == "branch" else " -> "
    return sep.join(annotate_step(s) for s in steps)


def load_recipes(recipes_dir: Path) -> list[dict]:
    """Load all TOML recipes and return sorted list of recipe metadata."""
    recipes = []
    for path in sorted(recipes_dir.glob("*.toml")):
        with open(path, "rb") as f:
            data = tomllib.load(f)
        meta = data.get("recipe", {})
        if not meta:
            continue
        slug = path.stem
        steps = data.get("steps", [])
        mode = meta.get("mode", "sequential")
        recipes.append({
            "slug": slug,
            "name": meta.get("name", slug),
            "mode": mode,
            "steps": len(steps),
            "effects": build_effects_chain(steps, mode),
        })
    return recipes


def generate_typescript(recipes: list[dict]) -> str:
    """Generate the TypeScript source for recipes.ts."""
    lines = [
        "/**",
        " * DO NOT EDIT -- this file is auto-generated by scripts/generate_worker_recipes.py",
        " * from recipes/*.toml. To update, run: python scripts/generate_worker_recipes.py",
        " */",
        "",
        "/** A rottengenizdat recipe definition. */",
        "export interface Recipe {",
        "  slug: string;",
        "  name: string;",
        '  mode: "sequential" | "branch";',
        "  steps: number;",
        "  effects: string;",
        "}",
        "",
        "/** All available recipes. */",
        "export const RECIPES: readonly Recipe[] = [",
    ]

    for r in recipes:
        slug = escape_ts_string(r["slug"])
        name = escape_ts_string(r["name"])
        mode = escape_ts_string(r["mode"])
        effects = escape_ts_string(r["effects"])
        lines.append(
            f'  {{ slug: "{slug}", name: "{name}", '
            f'mode: "{mode}", steps: {r["steps"]}, '
            f'effects: "{effects}" }},'
        )

    lines.append("] as const;")
    lines.append("")
    lines.append('const slugSet = new Set(RECIPES.map((r) => r.slug));')
    lines.append("")
    lines.append("/** Check whether a slug corresponds to a valid recipe. */")
    lines.append("export function isValidRecipe(slug: string): boolean {")
    lines.append("  return slugSet.has(slug);")
    lines.append("}")
    lines.append("")
    lines.append("/** Look up a recipe by slug, or undefined if not found. */")
    lines.append("export function getRecipe(slug: string): Recipe | undefined {")
    lines.append("  return RECIPES.find((r) => r.slug === slug);")
    lines.append("}")
    lines.append("")
    lines.append("/**")
    lines.append(" * Format the full recipe list for Slack display, grouped by mode.")
    lines.append(" */")
    lines.append("export function formatRecipeList(): string {")
    lines.append('  const sequential = RECIPES.filter((r) => r.mode === "sequential");')
    lines.append('  const branch = RECIPES.filter((r) => r.mode === "branch");')
    lines.append("")
    lines.append("  const fmt = (r: Recipe) => `\\`${r.slug}\\` \\u2014 ${r.name} (${r.steps} steps)`;")
    lines.append("")
    lines.append("  const sections: string[] = [];")
    lines.append("  if (sequential.length > 0) {")
    lines.append('    sections.push(`*Sequential (each step feeds the next):*\\n${sequential.map(fmt).join("\\n")}`);')
    lines.append("  }")
    lines.append("  if (branch.length > 0) {")
    lines.append('    sections.push(`*Branch (all steps run in parallel, then mix):*\\n${branch.map(fmt).join("\\n")}`);')
    lines.append("  }")
    lines.append("")
    lines.append('  return `*Available Recipes (${RECIPES.length} total):*\\n\\n${sections.join("\\n\\n")}`;')
    lines.append("}")
    lines.append("")
    lines.append("/**")
    lines.append(" * Find recipes whose slugs contain the given substring.")
    lines.append(' * Returns up to 5 matches for "did you mean?" suggestions.')
    lines.append(" */")
    lines.append("export function suggestRecipes(input: string): Recipe[] {")
    lines.append("  const lower = input.toLowerCase();")
    lines.append('  return RECIPES.filter((r) => r.slug.includes(lower)).slice(0, 5);')
    lines.append("}")
    lines.append("")

    return "\n".join(lines)


def main() -> None:
    parser = argparse.ArgumentParser(description="Generate worker/src/recipes.ts from TOML recipes")
    parser.add_argument("--recipes-dir", type=Path, default=None, help="Path to recipes directory")
    parser.add_argument("--output", type=Path, default=None, help="Output TypeScript file path")
    args = parser.parse_args()

    repo_root = find_repo_root()
    recipes_dir = args.recipes_dir or repo_root / "recipes"
    output_path = args.output or repo_root / "worker" / "src" / "recipes.ts"

    if not recipes_dir.is_dir():
        print(f"Error: recipes directory not found: {recipes_dir}", file=sys.stderr)
        sys.exit(1)

    recipes = load_recipes(recipes_dir)
    if not recipes:
        print("Error: no recipes found", file=sys.stderr)
        sys.exit(1)

    ts_source = generate_typescript(recipes)

    output_path.parent.mkdir(parents=True, exist_ok=True)
    output_path.write_text(ts_source)

    print(f"Generated {output_path} with {len(recipes)} recipes")


if __name__ == "__main__":
    main()
```

```typescript
/**
 * Dispatch the rottengenizdat GitHub Actions workflow via the REST API.
 * Returns true on success, false on failure.
 */
export async function dispatchWorkflow(
  env: Env,
  recipe: string,
  sampleSaleCount: number = 3,
  inputMode: string = "splice",
  audioUrls: string[] = [],
): Promise<boolean> {
  const inputs: Record<string, string> = {
    recipe,
    sample_sale_count: String(sampleSaleCount),
    input_mode: inputMode,
  };
  if (audioUrls.length > 0) {
    inputs.audio_urls = audioUrls.join(",");
  }

  const response = await fetch(
    "https://api.github.com/repos/A-U-Supply/rottengenizdat/actions/workflows/rottengenizdat.yml/dispatches",
    {
      method: "POST",
      headers: {
        Authorization: `Bearer ${env.GITHUB_TOKEN}`,
        Accept: "application/vnd.github+json",
        "User-Agent": "rottengenizdat-slash-command/1.0",
        "X-GitHub-Api-Version": "2022-11-28",
      },
      body: JSON.stringify({ ref: "main", inputs }),
    },
  );

  if (response.status !== 204) {
    const text = await response.text();
    console.error(`GitHub dispatch failed: ${response.status} ${text}`);
  }
  return response.status === 204;
}

/** Format a duration in seconds to a human-readable string. */
export function formatDuration(seconds: number): string {
  if (seconds < 60) return `${seconds}s`;
  const mins = Math.floor(seconds / 60);
  const secs = seconds % 60;
  return secs > 0 ? `${mins}m ${secs}s` : `${mins}m`;
}

/** Format a single workflow run for Slack display. */
export function formatRun(run: WorkflowRun): string {
  const statusEmoji =
    run.status === "completed"
      ? run.conclusion === "success"
        ? ":white_check_mark:"
        : ":x:"
      : run.status === "in_progress"
        ? ":hourglass_flowing_sand:"
        : ":clock1:";

  const label =
    run.status === "completed"
      ? run.conclusion === "success"
        ? "success"
        : (run.conclusion ?? "failed")
      : run.status;

  const started = new Date(run.run_started_at || run.created_at);
  const updated = new Date(run.updated_at);
  const durationSec = Math.round((updated.getTime() - started.getTime()) / 1000);
  const duration =
    run.status === "completed" ? ` in ${formatDuration(durationSec)}` : "";

  const timeAgo = Math.round((Date.now() - started.getTime()) / 60000);
  const when = timeAgo < 1 ? "just now" : `${timeAgo}m ago`;

  const trigger = run.event === "schedule" ? "scheduled" : "manual";

  return `${statusEmoji} *${label}* (${trigger}, ${when}${duration}) -- <${run.html_url}|logs>`;
}

 * Fetch recent rottengenizdat workflow runs from the GitHub Actions API.

  "January", "February", "March", "April", "May", "June",
  "July", "August", "September", "October", "November", "December",
];

/**
 * Fetch GitHub Actions usage for the current billing month.
 */
export async function fetchActionsUsage(env: Env): Promise<ActionsUsage | null> {
  try {
    const now = new Date();
    const year = now.getUTCFullYear();
    const month = now.getUTCMonth() + 1;

    const response = await fetch(
      `https://api.github.com/orgs/A-U-Supply/settings/billing/usage?year=${year}&month=${month}`,
      {
        headers: {
          Authorization: `Bearer ${env.GITHUB_TOKEN}`,
          Accept: "application/vnd.github+json",
          "User-Agent": "rottengenizdat-slash-command/1.0",
          "X-GitHub-Api-Version": "2022-11-28",
        },
      },
    );

    if (!response.ok) {
      console.error(`GitHub billing API failed: ${response.status} ${response.statusText}`);
      return null;
    }

    const data = (await response.json()) as { usageItems: UsageItem[] };
    const items = (data.usageItems ?? []).filter(
      (i) => i.product === "actions" && i.unitType === "Minutes",
    );

    let orgMinutes = 0;
    let rottengenezdatMinutes = 0;
    for (const item of items) {
      orgMinutes += item.quantity;
      if (item.repositoryName === "rottengenizdat") {
        rottengenezdatMinutes += item.quantity;
      }
    }

    return {
      orgMinutes: Math.round(orgMinutes),
      rottengenezdatMinutes: Math.round(rottengenezdatMinutes),
      includedMinutes: INCLUDED_MINUTES,
      month: MONTH_NAMES[month - 1],
    };
  } catch {
    return null;
  }
}
```

```typescript
/** Push a new view onto the Slack modal stack. */

/**
 * Handle Slack interactive payloads (button clicks, modal submissions).
 */
export async function handleInteraction(
  body: string,
  env: Env,
  ctx?: ExecutionContext,
): Promise<Response> {
  const params = new URLSearchParams(body);
  const payloadStr = params.get("payload");
  if (!payloadStr) {
    return new Response("Missing payload", { status: 400 });
  }

  let payload: SlackInteractionPayload;
  try {
    payload = JSON.parse(payloadStr);
  } catch {
    return new Response("Invalid JSON payload", { status: 400 });
  }

  switch (payload.type) {
    case "block_actions": {
      const actions = payload.actions || [];
      for (const action of actions) {
        switch (action.action_id) {
          case "modal_open_help": {
            if (payload.trigger_id) {
              const view = buildHelpView();
              await pushView(env, payload.trigger_id, view);
            }
            break;
          }
          case "modal_open_status": {
            if (payload.trigger_id) {
              const [runs, usage] = await Promise.all([
                fetchWorkflowRuns(env),
                fetchActionsUsage(env),
              ]);
              const view = buildStatusView(runs, usage);
              if (payload.view) {
                await pushView(env, payload.trigger_id, view);
              } else {
                await openView(env, payload.trigger_id, view);
              }
            }
            break;
          }
          default:
            break;
        }
      }
      return new Response("", { status: 200 });
    }

    case "view_submission": {
      if (payload.view?.callback_id === "rottengenizdat_run") {
        const vals = payload.view.state.values;
        const recipe =
          vals.recipe_block?.recipe_select?.selected_option?.value ?? null;
        const sampleCount = parseInt(
          vals.sample_count_block?.sample_count_select?.selected_option?.value ?? "3",
          10,
        );
        const inputMode =
          vals.input_mode_block?.input_mode_select?.selected_option?.value ?? "splice";
        const rawUrls: string = vals.urls_block?.audio_urls?.value ?? "";
        const urls = rawUrls
          .split(/[\s,]+/)
          .map((u: string) => u.trim())
          .filter((u: string) => /^https?:\/\//i.test(u));

        const resolvedRecipe =
          recipe && recipe !== "random" ? recipe : "";
        const recipeName =
          getRecipe(resolvedRecipe)?.name ?? (resolvedRecipe || "random");

        const channelId = payload.view.private_metadata;
        const userId = payload.user.id;

        // Dispatch in background — modal must respond immediately
        const work = (async () => {
          const [ok, usage] = await Promise.all([
            dispatchWorkflow(env, resolvedRecipe, sampleCount, inputMode, urls),
            fetchActionsUsage(env),
          ]);

          if (channelId) {
            const msg = ok
              ? `:radio: Firing up *${recipeName}* with ${sampleCount} sample(s) (${inputMode})... results incoming.`
              : `:warning: Failed to dispatch workflow.`;
            const blocks = ok ? confirmationBlocks(msg) : undefined;
            if (blocks && usage) blocks.push(buildUsageContextShort(usage));
            const body: Record<string, unknown> = { channel: channelId, user: userId, text: msg };
            if (blocks) body.blocks = blocks;
            await fetch("https://slack.com/api/chat.postEphemeral", {
              method: "POST",
              headers: {
                "Content-Type": "application/json",
                Authorization: `Bearer ${env.SLACK_BOT_TOKEN}`,
              },
              body: JSON.stringify(body),
            });
          }
        })();
        if (ctx) ctx.waitUntil(work);
        else work.catch(console.error);
      }
      return new Response(JSON.stringify({ response_action: "clear" }), {
        headers: { "Content-Type": "application/json" },
      });
    }

    case "block_suggestion": {
      return new Response(JSON.stringify({ options: [] }), {
        headers: { "Content-Type": "application/json" },
      });
    }

    default:
      return new Response("", { status: 200 });
  }
}
```

```typescript
      const msg = `:radio: Firing up *${command}*${urlNote}...`;
      const blocks: object[] = [
        { type: "section", text: { type: "mrkdwn", text: msg } },
        {
          type: "actions",
          elements: [{
            type: "button",
            text: { type: "plain_text", text: "Check Status" },
            action_id: "modal_open_status",
          }],
        },
      ];
      if (usage) blocks.push(buildUsageContextShort(usage));
      return new Response(
        JSON.stringify({ response_type: "ephemeral", text: msg, blocks }),
        { headers: { "Content-Type": "application/json" } },
      );
    }
    return slackResponse(":warning: Failed to dispatch workflow. Check the GitHub token configuration.");
  }

  // Invalid recipe -- suggest similar ones
  const suggestions = suggestRecipes(command);
  let errorText = `:x: Unknown recipe \`${command}\`.`;
  if (suggestions.length > 0) {
    const suggestionList = suggestions.map((r) => `\`${r.slug}\``).join(", ");
    errorText += `\n\nDid you mean: ${suggestionList}?`;
  }
  errorText += "\n\nUse `/rottengenizdat list` to see all available recipes.";
  return slackResponse(errorText);
}

async function openViewModal(env: Env, triggerId: string, view: object): Promise<void> {
  const resp = await fetch("https://slack.com/api/views.open", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${env.SLACK_BOT_TOKEN}`,
    },
    body: JSON.stringify({ trigger_id: triggerId, view }),
  });
  const data = (await resp.json()) as { ok: boolean; error?: string };
  if (!data.ok) {
    console.error(`views.open failed: ${data.error}`);
  }
}

async function openModal(env: Env, triggerId: string, channelId: string): Promise<void> {
  const view = buildModalView(channelId);
  await openViewModal(env, triggerId, view);
}

// ---------------------------------------------------------------------------
// Worker entry point
// ---------------------------------------------------------------------------

export default {
  async fetch(request: Request, env: Env, ctx: ExecutionContext): Promise<Response> {
    const url = new URL(request.url);

    if (request.method === "POST") {
      const body = await request.text();

      const valid = await verifySlackSignature(
        request,
        body,
        env.SLACK_SIGNING_SECRET,
      );
      if (!valid) {
        return new Response("Invalid signature", { status: 401 });
      }

      if (url.pathname === "/slack/interactions") {
        return handleInteraction(body, env, ctx);
      }

      if (url.pathname === "/slack/commands") {
        return handleSlashCommand(body, env, ctx);
      }
    }

    return new Response("Not Found", { status: 404 });
  },
};
```

```yaml
name: rottengenizdat

on:
  workflow_dispatch:
    inputs:
      recipe:
        description: 'Recipe name (leave empty for random)'
        default: ''
      sample_sale_count:
        description: 'Number of samples to pull from #sample-sale'
        default: '3'
      input_mode:
        description: 'Input combination mode: splice, concat, independent'
        default: 'splice'
      audio_urls:
        description: 'Comma-separated audio/video URLs (optional)'
        default: ''

jobs:
  rottengenizdat:
    runs-on: ubuntu-latest
    permissions:
      contents: read

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install system dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y ffmpeg

      - name: Install Python dependencies
        run: |
          pip install uv
          uv pip install --system -e ".[torch]"

      - name: Pick recipe
        id: recipe
        run: |
          RECIPE="${{ inputs.recipe }}"
          if [ -z "$RECIPE" ]; then
            RECIPE=$(ls recipes/*.toml | shuf -n1 | xargs -I{} basename {} .toml)
            echo "Randomly selected: $RECIPE"
          fi
          echo "name=$RECIPE" >> "$GITHUB_OUTPUT"

      - name: Run rottengenizdat
        env:
          SLACK_BOT_TOKEN: ${{ secrets.SLACK_BOT_TOKEN }}
        run: |
          CMD="rotten recipe run recipes/${{ steps.recipe.outputs.name }}.toml"
          CMD="$CMD --sample-sale-count ${{ inputs.sample_sale_count }}"
          if [ "${{ inputs.input_mode }}" != "splice" ]; then
            CMD="$CMD --mode ${{ inputs.input_mode }}"
          fi
          CMD="$CMD -o output.wav"
          echo "Running: $CMD"
          $CMD

      - name: Post result to Slack
        if: success()
        env:
          SLACK_BOT_TOKEN: ${{ secrets.SLACK_BOT_TOKEN }}
        run: |
          RECIPE="${{ steps.recipe.outputs.name }}"
          RUN_URL="${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}"

          # Upload the audio file
          UPLOAD_RESPONSE=$(curl -s -X POST https://slack.com/api/files.getUploadURLExternal \
            -H "Authorization: Bearer $SLACK_BOT_TOKEN" \
            -d "filename=rottengenizdat-${RECIPE}.wav" \
            -d "length=$(stat -c%s output.wav)")

          UPLOAD_URL=$(echo "$UPLOAD_RESPONSE" | python3 -c "import sys, json; print(json.load(sys.stdin)['upload_url'])")
          FILE_ID=$(echo "$UPLOAD_RESPONSE" | python3 -c "import sys, json; print(json.load(sys.stdin)['file_id'])")

          curl -s -X POST "$UPLOAD_URL" \
            -F "file=@output.wav"

          curl -s -X POST https://slack.com/api/files.completeUploadExternal \
            -H "Authorization: Bearer $SLACK_BOT_TOKEN" \
            -H "Content-Type: application/json" \
            -d "{\"files\": [{\"id\": \"$FILE_ID\", \"title\": \"rottengenizdat: ${RECIPE}\"}], \"channel_id\": \"sample-sale\", \"initial_comment\": \":radio: *rottengenizdat: ${RECIPE}* \\n<${RUN_URL}|View logs>\"}"

      - name: Notify Slack on failure
        if: failure()
        env:
          SLACK_BOT_TOKEN: ${{ secrets.SLACK_BOT_TOKEN }}
        run: |
          RECIPE="${{ steps.recipe.outputs.name }}"
          RUN_URL="${{ github.server_url }}/${{ github.repository }}/actions/runs/${{ github.run_id }}"
          curl -s -X POST https://slack.com/api/chat.postMessage \
            -H "Authorization: Bearer $SLACK_BOT_TOKEN" \
            -H "Content-Type: application/json" \
            -d "{\"channel\": \"sample-sale\", \"text\": \":x: rottengenizdat run failed (recipe: ${RECIPE}). <${RUN_URL}|View logs>\"}"
```
